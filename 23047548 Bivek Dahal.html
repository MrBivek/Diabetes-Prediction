    "Logistic Regression": {"Predictions": log_reg_predictions},
    "Random Forest": {"Predictions": rf_predictions},
    "XGBoost": {"Predictions": xgb_predictions}

# Looping through each model and plotting its confusion matrix
for ax, (name, res) in zip(axes, results.items()):
    cm = confusion_matrix(y_test, res["Predictions"]) 
    sns.heatmap(
        cm,
        annot=True,
        fmt="d",
        cmap="Blues",
        xticklabels=["No Diabetes", "Diabetes"],
        yticklabels=["No Diabetes", "Diabetes"],
        ax=ax
    )
    ax.set_title(name) # Setting title for each subplot
    ax.set_xlabel("Predicted") # Labeling x-axis
    ax.set_ylabel("Actual")    # Labeling y-axis
